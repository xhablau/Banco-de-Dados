CREATE TABLE users_matriz(
    id INT NOT NULL,
    cod_matriz VARCHAR (60) UNIQUE NOT NULL,
    login_matriz VARCHAR (60) NOT NULL,
    senha_matriz VARCHAR (60) NOT NULL,
    nome_matriz VARCHAR (50) NOT NULL ,
    cnpj_matriz VARCHAR (30) NOT NULL,
    email_matriz VARCHAR (50) NOT NULL,
    endereco_matriz VARCHAR (30) NOT NULL,
    numero_end_matriz INT NOT NULL,
    cep_matriz VARCHAR (30) NOT NULL,    
    comissao_matriz VARCHAR (20) NOT NULL,   
    porc_maior_saque_matriz BOOLEAN NOT NULL,
    porc_menor_saque_matriz BOOLEAN NOT NULL,
    porc_maior_venda_matriz BOOLEAN NOT NULL,
    porc_menor_venda_matriz BOOLEAN NOT NULL,
    porc_maior_parcelado_matriz BOOLEAN NOT NULL,
    porc_menor_parcelado_matriz BOOLEAN NOT NULL,
    valor_saque_matriz VARCHAR (20) NOT NULL,
    valor_venda_matriz VARCHAR (20) NOT NULL,
    valor_parcelado_matriz VARCHAR (20) NOT NULL,
    taxa_menor_venda_matriz VARCHAR (20) NOT NULL,
    taxa_maior_venda_matriz VARCHAR (20) NOT NULL,
    taxa_menor_saque_matriz VARCHAR (20) NOT NULL,
    taxa_maior_saque_matriz VARCHAR (20) NOT NULL,
    taxa_menor_parcelado_matriz VARCHAR (20) NOT NULL,
    taxa_maior_parcelado_matriz VARCHAR (20) NOT NULL,
    PRIMARY KEY (id, login_matriz, cnpj_matriz, cod_matriz), UNIQUE (id,login_matriz,cnpj_matriz)
);


CREATE TABLE users_filial(
    id INT NOT NULL,
    cod_matriz_filial VARCHAR (60) UNIQUE REFERENCES users_matriz (cod_matriz) NOT NULL,
    cod_filial VARCHAR (60) UNIQUE NOT NULL,
    login_filial VARCHAR (60) NOT NULL,
    senha_filial VARCHAR (60) NOT NULL,
    nome_filial VARCHAR (50) NOT NULL,
    cnpj_filial VARCHAR (30) NOT NULL,
    email_filial VARCHAR (50) NOT NULL,
    endereco_filial VARCHAR (30) NOT NULL,
    numero_end_filial INT NOT NULL,
    cep_filial VARCHAR (30) NOT NULL,
    agencia_filial VARCHAR (10) NOT NULL,
    conta_filial VARCHAR (20) NOT NULL,
    tipo_conta_filial VARCHAR (20) NOT NULL,
    compe_banco_filial INT NOT NULL,
    nome_banco_filial VARCHAR (30) NOT NULL,    
    comissao_filial VARCHAR (20) NOT NULL,
    porc_maior_saque_filial BOOLEAN NOT NULL,
    porc_menor_saque_filial BOOLEAN NOT NULL,
    porc_maior_venda_filial BOOLEAN NOT NULL,
    porc_menor_venda_filial BOOLEAN NOT NULL,
    porc_maior_parcelado_filial BOOLEAN NOT NULL,
    porc_menor_parcelado_filial BOOLEAN NOT NULL,
    valor_saque_filial VARCHAR (20) NOT NULL,
    valor_venda_filial VARCHAR (20) NOT NULL,
    valor_parcelado_filial VARCHAR (20) NOT NULL,
    taxa_menor_venda_filial VARCHAR (20) NOT NULL,
     taxa_maior_venda_filial VARCHAR (20) NOT NULL,
     taxa_menor_saque_filial VARCHAR (20) NOT NULL,
     taxa_maior_saque_filial VARCHAR (20) NOT NULL,
     taxa_menor_parcelado_filial VARCHAR (20) NOT NULL,
     taxa_maior_parcelado_filial VARCHAR (20) NOT NULL,
    PRIMARY KEY (id, cod_matriz_filial,login_filial, cnpj_filial, cod_filial), UNIQUE (id,cod_matriz_filial,login_filial,cnpj_filial));

CREATE TABLE users_operador(
    id INT NOT NULL ,
    cod_matriz_operador VARCHAR (60) UNIQUE REFERENCES users_matriz (cod_matriz) NOT NULL,
    cod_filial_operador VARCHAR (60) UNIQUE REFERENCES users_filial (cod_filial) NOT NULL,
    cod_operador VARCHAR (60) UNIQUE NOT NULL,
    login_operador VARCHAR (60) NOT NULL,
    senha_operador VARCHAR (60) NOT NULL,
    nome_operador VARCHAR (50) NOT NULL,
    email_operador VARCHAR (50) NOT NULL,    
    porc_maior_saque_op BOOLEAN NOT NULL,
    porc_menor_saque_op BOOLEAN NOT NULL,
    porc_maior_venda_op BOOLEAN NOT NULL,
    porc_menor_venda_op BOOLEAN NOT NULL,
    porc_maior_parcelado_op BOOLEAN NOT NULL,
    porc_menor_parcelado_op BOOLEAN NOT NULL,
    valor_saque_operador VARCHAR (20) NOT NULL,
    valor_venda_operador VARCHAR (20) NOT NULL,
    valor_parcelado_operador VARCHAR (20) NOT NULL,
    taxa_menor_venda_operador VARCHAR (20) NOT NULL,
    taxa_maior_venda_operador VARCHAR (20) NOT NULL,
    taxa_menor_saque_operador VARCHAR (20) NOT NULL,
    taxa_maior_saque_operador VARCHAR (20) NOT NULL,
    taxa_menor_parcelado_operador VARCHAR (20) NOT NULL,
    taxa_maior_parcelado_operador VARCHAR (20) NOT NULL,
    PRIMARY KEY (id, cod_operador,login_operador, cod_matriz_operador, cod_filial_operador), UNIQUE (id,cod_operador,login_operador)
    );
